Female Squat

Load in appropriate packages and datasets:

library(tidyverse)
library(readr)
library(ggplot2)
library(dunn.test)
library(FSA)
library(reshape2)

openpl_male <- read_csv("~/Desktop/SYE/github repos/stat450-spr2026-score/keelinjoyce/powerlifting//openpl_male.csv")

openpl_female <- read_csv("~/Desktop/SYE/github repos/stat450-spr2026-score/keelinjoyce/powerlifting//openpl_female.csv")

Arrange data appropriately:

female_squat_ranked <- openpl_female |> arrange(desc(Squat)) |>
  mutate(rank = row_number()) |>
  filter(!is.na(AgeClass)) |>
  mutate(AgeClass = fct_relevel(AgeClass, "5-12")) |>
  select(Name, Age, AgeClass, Squat, rank)|>
  mutate(AgeClass = fct_recode(AgeClass,
                              "05-12" = "5-12"))
## recoding "5-12" as "05-12" so it appears in the correct order

female_squat_ranked |>
  group_by(AgeClass)  |>
  summarise(n())

Look at distribution of the data:

ggplot(data = female_squat_ranked, aes(x = AgeClass, y = Squat)) +
  geom_boxplot() +
  theme_minimal()

Run the test, confirm significant p-value:

female_squat_results <- kruskal.test(female_squat_ranked$rank ~ 
                                       female_squat_ranked$AgeClass)
female_squat_results

Run pairwise test:

female_squat_dunn <- dunn.test(female_squat_ranked$rank, female_squat_ranked$AgeClass, method = "bh")
dunn_results_df <- as.data.frame(female_squat_dunn$P.adjusted, female_squat_dunn$comparisons) |>
